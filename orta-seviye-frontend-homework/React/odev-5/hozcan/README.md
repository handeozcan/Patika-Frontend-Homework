## Install

```bash
npm install --save hozcan
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'hozcan'
import 'hozcan/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
